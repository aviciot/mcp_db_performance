with pop as (  
 select oo.id,oo.payment_date
        from 
        OWS.MERCHANT_STATEMENT oo
       join 
         (
           select o.contract_id, o.payment_date
           from OWS.MERCHANT_STATEMENT o
           join 
           (
           select distinct ow.contract_id, ow.payment_date
           from OWS.MERCHANT_STATEMENT ow
           where 
           ow.statement_status='R'
    and ow.contract_id in($$CONTRACTS_LIST1 $$CONTRACTS_LIST2 $$CONTRACTS_LIST3 $$CONTRACTS_LIST4 $$CONTRACTS_LIST5 -99)
           and  ow.READY_DATE>to_date('$$ETL_RUN_DATE','MM/DD/YYYY HH24:MI:SS') AND ow.READY_DATE < sysdate ) b 
           
           
           on b.contract_id=o.contract_id and b.payment_date=o.payment_date
             where  b.contract_id in ($$CONTRACTS_LIST1 $$CONTRACTS_LIST2 $$CONTRACTS_LIST3 $$CONTRACTS_LIST4 $$CONTRACTS_LIST5 -99) 
           group by o.contract_id, o.payment_date
           having 
           count(o.payment_id)= sum (case when o.statement_status='R' then 1 else 0 end )
         
          ) a
            on a.payment_date=oo.payment_date and a.contract_id=oo.contract_id
   and oo.contract_id in ($$CONTRACTS_LIST1 $$CONTRACTS_LIST2 $$CONTRACTS_LIST3 $$CONTRACTS_LIST4 $$CONTRACTS_LIST5 -99) 
),
 min_payment_date as (
select min(payment_date) as payment_date from pop
),
  
org_pop as
(SELECT /*+ use_nl(pop,OA)  */
  distinct 
  OA.TRANS_DOC_ID, MTT.ID
    from pop
   inner join OWS.MERCHANT_STATEMENT_ENTRY OA
      on pop.id = oa.merchant_statement__oid
	  LEFT JOIN ows.M_TRANSACTION MTT ON 
                           MTT.ID = OA.TRANS_MTR_ID 
                           AND 
                           MTT.TRANS_CODE = 'CCCACA'
      where
      OA.payment_date >= (select min_payment_date.payment_date from min_payment_date)
       
      ),
